if __name__ == '__main__':
    import sys
    app = QtWidgets.QApplication(sys.argv)
    MainWindow = QtWidgets.QTabWidget()
    ui = Ui_TabWidget() #this may not be TabWidget, it depends.
    ui.setupUi(MainWindow)
    MainWindow.show()
    sys.exit(app.exec_())
